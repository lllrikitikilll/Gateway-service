Table users {
  id integer [primary key]
  username varchar unique
  hash_password varchar
  verify bool
  balance integer
}

Table tokens {
  id integer [primary key]
  user_id integer
  token varchar
  created_at timestamp
  expired timestamp
}

Table Transactions {
  id integer [primary key]
  user_id integer
  operation varchar
  timestamp timestamp
  amount integer
  label_id integer
}

Table labels {
  id integer [primary key]
  label_name varchar unique
}

Ref: tokens.user_id > users.id
Ref: Transactions.user_id > users.id
Ref: Transactions.label_id <> labels.id
